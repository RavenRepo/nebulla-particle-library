import { CurlNoiseParticles } from './src/CurlNoiseParticles.js';

// Create particle system in a specific container
const particles = new CurlNoiseParticles({
    containerElement: document.getElementById('particle-container'),
    particleCount: 100000,
    enableBokeh: true,
    color1: 0x0066ff,
    color2: 0xff00ff,
    pointSize: 2.5,
    curlStrength: 2.5,
});

// Update configuration in real-time
particles.updateConfig({
    curlStrength: 3.5,
    noiseSpeed: 0.5,
});

// Clean up when done
particles.dispose();
